#!/usr/bin/env python3
"""
Simple Phone Detection - No PyTorch needed
Using traditional computer vision methods
"""

import cv2
import numpy as np
import time
import os

print("Simple Phone Detection")
print("=" * 50)

def main():
    # Open camera
    cap = cv2.VideoCapture(0)
    if not cap.isOpened():
        print("Error: Cannot open camera")
        return
  
    cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
    cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
  
    print("Camera opened")
    print("Press 'q' to quit, 's' to save image")
    print("-" * 50)
  
    frame_count = 0
    start_time = time.time()
  
    while True:
        ret, frame = cap.read()
        if not ret:
            break
      
        frame_count += 1
        frame = cv2.flip(frame, 1)
      
        # Convert to HSV color space
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
      
        # Detect dark objects (phones are usually dark)
        lower_dark = np.array([0, 0, 0])
        upper_dark = np.array([180, 255, 80])
        mask = cv2.inRange(hsv, lower_dark, upper_dark)
      
        # Morphological operations
        kernel = np.ones((5, 5), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
      
        # Find contours
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
      
        detections = []
        for contour in contours:
            area = cv2.contourArea(contour)
            if 1000 < area < 30000:
                x, y, w, h = cv2.boundingRect(contour)
                aspect_ratio = h / w
              
                # Phones are usually taller than wide, aspect ratio 1.5-3.5
                if 1.5 < aspect_ratio < 3.5:
                    detections.append((x, y, w, h, area))
      
        # Draw detection results
        for (x, y, w, h, area) in detections:
            cv2.rectangle(frame, (x, y), (x+w, y+h), (0, 255, 0), 2)
            cv2.putText(frame, f"Phone", (x, y-10), 
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)
      
        # Calculate FPS
        elapsed = time.time() - start_time
        fps = frame_count / elapsed if elapsed > 0 else 0
      
        # Display information
        cv2.putText(frame, f"FPS: {fps:.1f}", (10, 30),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 0), 2)
        cv2.putText(frame, f"Detections: {len(detections)}", (10, 60),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 0), 2)
        cv2.putText(frame, "Simple Detector (No PyTorch)", (10, 90),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.6, (100, 255, 100), 2)
        cv2.putText(frame, "Press 'q' to quit, 's' to save", (10, 120),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 255), 1)
      
        cv2.imshow('Phone Detection (Simple)', frame)
      
        key = cv2.waitKey(1) & 0xFF
        if key == ord('q'):
            break
        elif key == ord('s'):
            timestamp = time.strftime("%Y%m%d_%H%M%S")
            filename = f"phone_detection_{timestamp}.jpg"
            cv2.imwrite(filename, frame)
            print(f"Image saved: {filename}")
  
    cap.release()
    cv2.destroyAllWindows()
  
    print(f"\nRun statistics:")
    print(f"Total frames: {frame_count}")
    print(f"Average FPS: {fps:.1f}")

#!/usr/bin/env python3
"""
Complete Phone Detection - Using YOLOv5
"""

import cv2
import torch
import numpy as np
import time

print("Complete Phone Detection - YOLOv5")
print("=" * 50)

def main():
    print("Loading YOLOv5 model...")
    try:
        # Load model
        model = torch.hub.load('ultralytics/yolov5', 'yolov5s', pretrained=True)
        model.conf = 0.25  # Confidence threshold
      
        # Find phone class IDs
        phone_ids = []
        for idx, name in model.names.items():
            if 'cell phone' in name.lower() or 'phone' in name.lower():
                phone_ids.append(idx)
                print(f"Found phone class: {name} (ID: {idx})")
      
        if not phone_ids:
            phone_ids = [67]  # Default phone class ID
            print(f"Using default phone class ID: 67")
      
        print("Model loaded successfully")
    except Exception as e:
        print(f"Model loading failed: {e}")
        print("Please ensure PyTorch is correctly installed")
        return
  
    # Open camera
    cap = cv2.VideoCapture(0)
    if not cap.isOpened():
        print("Error: Cannot open camera")
        return
  
    cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
    cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
  
    print("Camera opened")
    print("Press 'q' to quit, 's' to save image")
    print("-" * 50)
  
    frame_count = 0
    detection_count = 0
    start_time = time.time()
  
    while True:
        ret, frame = cap.read()
        if not ret:
            break
      
        frame_count += 1
        frame = cv2.flip(frame, 1)
      
        # YOLO detection
        results = model(frame, size=320)
        detections = results.pandas().xyxy[0]
      
        # Filter phone detections
        phone_detections = []
        for _, det in detections.iterrows():
            if det['class'] in phone_ids and det['confidence'] >= 0.25:
                x1, y1, x2, y2 = int(det['xmin']), int(det['ymin']), int(det['xmax']), int(det['ymax'])
                phone_detections.append((x1, y1, x2, y2, det['confidence']))
                detection_count += 1
      
        # Draw detection results
        for (x1, y1, x2, y2, conf) in phone_detections:
            cv2.rectangle(frame, (x1, y1), (x2, y2), (0, 255, 0), 2)
            label = f"Phone: {conf:.2f}"
            cv2.putText(frame, label, (x1, y1-10), 
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)
      
        # Calculate FPS
        elapsed = time.time() - start_time
        fps = frame_count / elapsed if elapsed > 0 else 0
      
        # Display information
        cv2.putText(frame, f"FPS: {fps:.1f}", (10, 30),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 0), 2)
        cv2.putText(frame, f"Detections: {len(phone_detections)}", (10, 60),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 0), 2)
        cv2.putText(frame, "YOLOv5 Detector", (10, 90),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.6, (100, 255, 100), 2)
        cv2.putText(frame, "Press 'q' to quit, 's' to save", (10, 120),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 255), 1)
      
        cv2.imshow('Phone Detection (YOLOv5)', frame)
      
        key = cv2.waitKey(1) & 0xFF
        if key == ord('q'):
            break
        elif key == ord('s'):
            timestamp = time.strftime("%Y%m%d_%H%M%S")
            filename = f"phone_detection_yolo_{timestamp}.jpg"
            cv2.imwrite(filename, frame)
            print(f"Image saved: {filename}")
  
    cap.release()
    cv2.destroyAllWindows()
  
    print(f"\nRun statistics:")
    print(f"Total frames: {frame_count}")
    print(f"Detection count: {detection_count}")
    print(f"Average FPS: {fps:.1f}")

if __name__ == "__main__":
    main()
